<script>
const APP_ID = 'KIN-028';

let pages = [];
let activePageIndex = -1;
let currentTool = 'adjust';
let cameraStream = null;
let facingMode = 'environment';
let cropState = null;
let draggingHandle = null;

// Image cache so slider drags don't re-decode the image
let editorImgCache = null;
let editorImgCacheSrc = null;

const isDemo = new URLSearchParams(location.search).get('demo') === 'true';

function init() {
  loadTheme();
  if (isDemo) {
    document.getElementById('demoBar').classList.remove('hidden');
    loadDemoData();
  }
  render();
}

function loadDemoData() {
  const colors = ['#d4c5a9', '#b8c4d4', '#c9d4b8'];
  const labels = ['Receipt', 'Contract p1', 'Contract p2'];
  for (let i = 0; i < 3; i++) {
    const c = document.createElement('canvas');
    c.width = 600; c.height = 848;
    const ctx = c.getContext('2d');
    ctx.fillStyle = colors[i]; ctx.fillRect(0, 0, 600, 848);
    ctx.fillStyle = '#555'; ctx.font = '16px sans-serif';
    ctx.fillText(labels[i], 40, 60);
    for (let j = 0; j < 12; j++) {
      ctx.fillStyle = '#999';
      ctx.fillRect(40, 90 + j * 30, 200 + Math.random() * 300, 8);
    }
    pages.push({
      imageData: c.toDataURL('image/jpeg', 0.85),
      adjustments: { brightness: 0, contrast: 0, bw: false },
      crop: null
    });
  }
  activePageIndex = 0;
}

function startFresh() {
  pages = []; activePageIndex = -1;
  editorImgCache = null; editorImgCacheSrc = null;
  document.getElementById('demoBar').classList.add('hidden');
  history.replaceState(null, '', location.pathname);
  render();
}

function loadTheme() {
  const dark = localStorage.getItem('kin-scan-theme') === 'dark';
  if (dark) {
    document.body.classList.add('dark');
    document.querySelector('meta[name="theme-color"]').content = '#111111';
    document.getElementById('themeBtn').textContent = '🌙';
  }
}

function toggleTheme() {
  const isDark = document.body.classList.toggle('dark');
  localStorage.setItem('kin-scan-theme', isDark ? 'dark' : 'light');
  document.querySelector('meta[name="theme-color"]').content = isDark ? '#111111' : '#ffffff';
  document.getElementById('themeBtn').textContent = isDark ? '🌙' : '☀️';
}

// =====================
// RENDER
// =====================
function render() {
  const main = document.getElementById('mainContent');

  if (pages.length === 0) {
    main.innerHTML = `
      <div class="empty-state">
        <div class="empty-icon">📄</div>
        <p>Scan documents to PDF.<br>Use your camera or import an image.</p>
        <div style="display:flex;gap:8px;justify-content:center;flex-wrap:wrap">
          <button class="btn-primary" onclick="openCamera()">📷 Scan</button>
          <button class="btn-secondary" onclick="importImage()">Import image</button>
        </div>
      </div>`;
    return;
  }

  let html = '<div class="pages-grid">';
  pages.forEach((p, i) => {
    const isFirst = i === 0;
    const isLast = i === pages.length - 1;
    html += `
      <div class="page-thumb ${i === activePageIndex ? 'active' : ''}" onclick="selectPage(${i})">
        <img src="${p.imageData}" alt="Page ${i + 1}">
        <span class="page-count">${i + 1}</span>
        <div class="page-actions">
          <button class="page-action-btn" onclick="event.stopPropagation();deletePage(${i})" title="Delete">✕</button>
        </div>
        ${pages.length > 1 ? `
        <div class="page-move-row">
          <button class="page-move-btn" onclick="event.stopPropagation();movePage(${i},-1)" ${isFirst ? 'disabled' : ''} title="Move left">◀</button>
          <button class="page-move-btn" onclick="event.stopPropagation();movePage(${i},1)" ${isLast ? 'disabled' : ''} title="Move right">▶</button>
        </div>` : ''}
      </div>`;
  });
  html += `
    <button class="add-page-card" onclick="showAddOptions()">
      <span class="plus">+</span>Add page
    </button></div>`;

  if (activePageIndex >= 0 && activePageIndex < pages.length) {
    const page = pages[activePageIndex];
    html += `
      <div class="editor">
        <div class="toolbar">
          <button class="${currentTool === 'adjust' ? 'active' : ''}" onclick="setTool('adjust')">Adjust</button>
          <button class="${currentTool === 'crop' ? 'active' : ''}" onclick="setTool('crop')">Crop</button>
          <button onclick="rotatePage()">Rotate</button>
          <button onclick="toggleBW()">${page.adjustments.bw ? 'B&W ✓' : 'B&W'}</button>
        </div>
        <div class="editor-canvas-wrap" id="canvasWrap">
          <canvas id="editorCanvas"></canvas>
        </div>`;

    if (currentTool === 'adjust') {
      const hasChanges = page.adjustments.brightness !== 0 || page.adjustments.contrast !== 0;
      html += `
        <div class="adjustments">
          <div class="slider-group">
            <label>Brightness</label>
            <input type="range" min="-100" max="100" value="${page.adjustments.brightness}" id="brightnessSlider" oninput="onBrightness(this.value)">
            <span class="slider-val" id="brightnessVal">${page.adjustments.brightness}</span>
          </div>
          <div class="slider-group">
            <label>Contrast</label>
            <input type="range" min="-100" max="100" value="${page.adjustments.contrast}" id="contrastSlider" oninput="onContrast(this.value)">
            <span class="slider-val" id="contrastVal">${page.adjustments.contrast}</span>
          </div>
          ${hasChanges ? '<div style="text-align:right"><button class="slider-reset" onclick="resetAdjustments()">Reset adjustments</button></div>' : ''}
        </div>`;
    }

    if (currentTool === 'crop' && page.crop) {
      html += `<div style="text-align:center;margin-top:8px">
        <button class="btn-secondary" onclick="applyCrop()" style="margin-right:8px">Apply crop</button>
        <button class="btn-danger" onclick="resetCrop()">Reset</button>
      </div>`;
    }
    html += '</div>';
  }

  html += `
    <div class="action-row">
      <button class="btn-primary" onclick="exportPDF()">
        Export PDF (${pages.length} page${pages.length !== 1 ? 's' : ''})
      </button>
    </div>`;

  main.innerHTML = html;

  if (activePageIndex >= 0) {
    const src = pages[activePageIndex].imageData;
    if (editorImgCacheSrc !== src) {
      editorImgCache = null; editorImgCacheSrc = null;
    }
    requestAnimationFrame(() => drawEditor());
  }
}

function movePage(index, direction) {
  const newIndex = index + direction;
  if (newIndex < 0 || newIndex >= pages.length) return;
  const temp = pages[index];
  pages[index] = pages[newIndex];
  pages[newIndex] = temp;
  if (activePageIndex === index) activePageIndex = newIndex;
  else if (activePageIndex === newIndex) activePageIndex = index;
  render();
}

async function openCamera() {
  try {
    cameraStream = await navigator.mediaDevices.getUserMedia({
      video: { facingMode, width: { ideal: 2048 }, height: { ideal: 2048 } }
    });
    showCameraUI();
  } catch (e) {
    alert('Camera access denied or not available. Try importing an image instead.');
  }
}

function showCameraUI() {
  const overlay = document.createElement('div');
  overlay.className = 'camera-overlay'; overlay.id = 'cameraOverlay';
  overlay.innerHTML = `
    <video autoplay playsinline muted id="cameraVideo"></video>
    <div class="camera-controls">
      <button class="camera-close" onclick="closeCamera()">✕</button>
      <button class="shutter-btn" onclick="capturePhoto()"></button>
      <button class="camera-switch" onclick="switchCamera()">🔄</button>
    </div>`;
  document.body.appendChild(overlay);
  document.getElementById('cameraVideo').srcObject = cameraStream;
}

function closeCamera() {
  if (cameraStream) { cameraStream.getTracks().forEach(t => t.stop()); cameraStream = null; }
  document.getElementById('cameraOverlay')?.remove();
}

async function switchCamera() {
  facingMode = facingMode === 'environment' ? 'user' : 'environment';
  closeCamera(); await openCamera();
}

function capturePhoto() {
  const video = document.getElementById('cameraVideo');
  const c = document.createElement('canvas');
  const maxDim = 2000;
  let w = video.videoWidth, h = video.videoHeight;
  if (Math.max(w, h) > maxDim) {
    const s = maxDim / Math.max(w, h);
    w = Math.round(w * s); h = Math.round(h * s);
  }
  c.width = w; c.height = h;
  c.getContext('2d').drawImage(video, 0, 0, w, h);
  closeCamera();
  addPage(c.toDataURL('image/jpeg', 0.9));
}

function importImage() {
  const input = document.createElement('input');
  input.type = 'file'; input.accept = 'image/*'; input.multiple = true;
  input.onchange = (e) => {
    Array.from(e.target.files).forEach(file => {
      const reader = new FileReader();
      reader.onload = (ev) => {
        const img = new Image();
        img.onload = () => {
          const c = document.createElement('canvas');
          const maxDim = 2000;
          let w = img.width, h = img.height;
          if (Math.max(w, h) > maxDim) {
            const s = maxDim / Math.max(w, h);
            w = Math.round(w * s); h = Math.round(h * s);
          }
          c.width = w; c.height = h;
          c.getContext('2d').drawImage(img, 0, 0, w, h);
          addPage(c.toDataURL('image/jpeg', 0.9));
        };
        img.src = ev.target.result;
      };
      reader.readAsDataURL(file);
    });
  };
  input.click();
}

function rotatePage() {

function drawEditor() {
  const canvas = document.getElementById('editorCanvas');
  if (!canvas || activePageIndex < 0) return;
  const page = pages[activePageIndex];

  const paint = (img) => {
    const wrap = document.getElementById('canvasWrap');
    if (!wrap) return;
    const displayW = wrap.clientWidth;
    const scale = displayW / img.width;
    canvas.width = img.width; canvas.height = img.height;
    canvas.style.width = displayW + 'px';
    canvas.style.height = (img.height * scale) + 'px';

    const ctx = canvas.getContext('2d');
    const b = 100 + page.adjustments.brightness;
    const c = 100 + page.adjustments.contrast;
    let f = `brightness(${b}%) contrast(${c}%)`;
    if (page.adjustments.bw) f += ' grayscale(100%)';
    ctx.filter = f;
    ctx.drawImage(img, 0, 0);
    ctx.filter = 'none';

    if (currentTool === 'crop' && page.crop) setupCropOverlay(canvas, page.crop, scale);
    else removeCropOverlay();
  };

  if (editorImgCache && editorImgCacheSrc === page.imageData) {
    paint(editorImgCache);
  } else {
    const img = new Image();
    img.onload = () => {
      editorImgCache = img; editorImgCacheSrc = page.imageData;
      paint(img);
    };
    img.src = page.imageData;
  }
}

// Fast path for slider dragging: uses cached image, throttled via rAF
let rafPending = false;
function drawEditorFast() {
  if (rafPending) return;
  rafPending = true;
  requestAnimationFrame(() => {
    rafPending = false;
    const canvas = document.getElementById('editorCanvas');
    if (!canvas || activePageIndex < 0 || !editorImgCache) return;
    const page = pages[activePageIndex];
    const ctx = canvas.getContext('2d');
    const b = 100 + page.adjustments.brightness;
    const c = 100 + page.adjustments.contrast;
    let f = `brightness(${b}%) contrast(${c}%)`;
    if (page.adjustments.bw) f += ' grayscale(100%)';
    ctx.filter = f;
    ctx.drawImage(editorImgCache, 0, 0);
    ctx.filter = 'none';
  });
}

// CROP
// =====================
function setupCropOverlay(canvas, crop, scale) {
  removeCropOverlay();
  const wrap = document.getElementById('canvasWrap');
  const overlay = document.createElement('div');
  overlay.className = 'crop-overlay'; overlay.id = 'cropOverlay';

  const cw = canvas.width * scale, ch = canvas.height * scale;
  const rx = crop.x * cw, ry = crop.y * ch;
  const rw = crop.w * cw, rh = crop.h * ch;

  overlay.innerHTML = `
    <div class="crop-rect" style="left:${rx}px;top:${ry}px;width:${rw}px;height:${rh}px"></div>
    <div class="crop-handle" data-corner="tl" style="left:${rx}px;top:${ry}px"></div>
    <div class="crop-handle" data-corner="tr" style="left:${rx + rw}px;top:${ry}px"></div>
    <div class="crop-handle" data-corner="bl" style="left:${rx}px;top:${ry + rh}px"></div>
    <div class="crop-handle" data-corner="br" style="left:${rx + rw}px;top:${ry + rh}px"></div>`;
  wrap.appendChild(overlay);

  overlay.querySelectorAll('.crop-handle').forEach(h => {
    const onStart = (e) => {
      e.preventDefault();
      draggingHandle = h.dataset.corner;
      const pt = getPointerXY(e);
      cropState = { startX: pt.x, startY: pt.y, origCrop: { ...crop } };

      const onMove = (ev) => {
        ev.preventDefault();
        const p = getPointerXY(ev);
        const dx = (p.x - cropState.startX) / cw;
        const dy = (p.y - cropState.startY) / ch;
        const oc = cropState.origCrop;
        if (draggingHandle === 'tl') {
          crop.x = Math.max(0, Math.min(oc.x + dx, oc.x + oc.w - 0.05));
          crop.y = Math.max(0, Math.min(oc.y + dy, oc.y + oc.h - 0.05));
          crop.w = oc.w - (crop.x - oc.x); crop.h = oc.h - (crop.y - oc.y);
        } else if (draggingHandle === 'tr') {
          crop.w = Math.max(0.05, Math.min(oc.w + dx, 1 - oc.x));
          crop.y = Math.max(0, Math.min(oc.y + dy, oc.y + oc.h - 0.05));
          crop.h = oc.h - (crop.y - oc.y);
        } else if (draggingHandle === 'bl') {
          crop.x = Math.max(0, Math.min(oc.x + dx, oc.x + oc.w - 0.05));
          crop.w = oc.w - (crop.x - oc.x);
          crop.h = Math.max(0.05, Math.min(oc.h + dy, 1 - oc.y));
        } else if (draggingHandle === 'br') {
          crop.w = Math.max(0.05, Math.min(oc.w + dx, 1 - oc.x));
          crop.h = Math.max(0.05, Math.min(oc.h + dy, 1 - oc.y));
        }
        updateCropVisuals(cw, ch, crop);
      };
      const onEnd = () => {
        draggingHandle = null;
        document.removeEventListener('mousemove', onMove);
        document.removeEventListener('mouseup', onEnd);
        document.removeEventListener('touchmove', onMove);
        document.removeEventListener('touchend', onEnd);
      };
      document.addEventListener('mousemove', onMove, { passive: false });
      document.addEventListener('mouseup', onEnd);
      document.addEventListener('touchmove', onMove, { passive: false });
      document.addEventListener('touchend', onEnd);
    };
    h.addEventListener('mousedown', onStart);
    h.addEventListener('touchstart', onStart, { passive: false });
  });
}

function getPointerXY(e) {
  if (e.touches && e.touches.length) return { x: e.touches[0].clientX, y: e.touches[0].clientY };
  return { x: e.clientX, y: e.clientY };
}

function updateCropVisuals(cw, ch, crop) {
  const overlay = document.getElementById('cropOverlay');
  if (!overlay) return;
  const rect = overlay.querySelector('.crop-rect');
  const handles = overlay.querySelectorAll('.crop-handle');
  const rx = crop.x * cw, ry = crop.y * ch, rw = crop.w * cw, rh = crop.h * ch;
  rect.style.left = rx + 'px'; rect.style.top = ry + 'px';
  rect.style.width = rw + 'px'; rect.style.height = rh + 'px';
  handles.forEach(h => {
    const c = h.dataset.corner;
    if (c === 'tl') { h.style.left = rx + 'px'; h.style.top = ry + 'px'; }
    if (c === 'tr') { h.style.left = (rx + rw) + 'px'; h.style.top = ry + 'px'; }
    if (c === 'bl') { h.style.left = rx + 'px'; h.style.top = (ry + rh) + 'px'; }
    if (c === 'br') { h.style.left = (rx + rw) + 'px'; h.style.top = (ry + rh) + 'px'; }
  });
}

function removeCropOverlay() { document.getElementById('cropOverlay')?.remove(); }

function applyCrop() {
  if (activePageIndex < 0) return;
  const page = pages[activePageIndex];
  if (!page.crop) return;
  const img = new Image();
  img.onload = () => {
    const sx = page.crop.x * img.width, sy = page.crop.y * img.height;
    const sw = page.crop.w * img.width, sh = page.crop.h * img.height;
    const c = document.createElement('canvas');
    c.width = Math.round(sw); c.height = Math.round(sh);
    c.getContext('2d').drawImage(img, sx, sy, sw, sh, 0, 0, c.width, c.height);
    page.imageData = c.toDataURL('image/jpeg', 0.9);
    page.crop = null; currentTool = 'adjust';
    editorImgCache = null; editorImgCacheSrc = null;
    render();
  };
  img.src = page.imageData;
}

function resetCrop() {
  if (activePageIndex < 0) return;
  pages[activePageIndex].crop = null;
  render();
}

function exportPDF() {
  if (pages.length === 0) return;
  const now = new Date();
  const dateStr = `${now.getFullYear()}-${String(now.getMonth()+1).padStart(2,'0')}-${String(now.getDate()).padStart(2,'0')}`;

  const backdrop = document.createElement('div');
  backdrop.className = 'modal-backdrop'; backdrop.id = 'exportModal';
  backdrop.innerHTML = `
    <div class="modal-content">
      <h2>Export PDF</h2>
      <input type="text" class="filename-input" id="filenameInput" value="scan-${dateStr}" placeholder="Filename">
      <div class="modal-hint" style="font-size:12px;color:var(--text-muted);margin-bottom:16px">.pdf will be added automatically</div>
      <div class="page-size-row">
        <label>Page size</label>
        <select id="pageSizeSelect">
          <option value="a4" selected>A4</option>
          <option value="letter">US Letter</option>
          <option value="fit">Fit to image</option>
        </select>
      </div>
      <div class="modal-actions">
        <button class="btn-secondary" onclick="closeExportModal()">Cancel</button>
        <button class="btn-primary" onclick="doExport()">Download</button>
      </div>
    </div>`;
  document.body.appendChild(backdrop);
  document.getElementById('filenameInput').focus();
  document.getElementById('filenameInput').select();
}

function closeExportModal() { document.getElementById('exportModal')?.remove(); }

async function doExport() {
  const filename = (document.getElementById('filenameInput').value || 'scan').trim();
  const pageSize = document.getElementById('pageSizeSelect').value;
  closeExportModal();

  const { jsPDF } = window.jspdf;
  const sizes = { a4: [210, 297], letter: [215.9, 279.4] };
  let pdf = null;

  for (let i = 0; i < pages.length; i++) {
    const processed = await processPageForExport(pages[i]);
    const img = await loadImage(processed);
    const imgAspect = img.width / img.height;
    let pW, pH;
    if (pageSize === 'fit') {
      pW = (img.width / 150) * 25.4; pH = (img.height / 150) * 25.4;
    } else { [pW, pH] = sizes[pageSize]; }

    if (i === 0) {
      pdf = new jsPDF({ orientation: pW > pH ? 'landscape' : 'portrait', unit: 'mm', format: pageSize === 'fit' ? [pW, pH] : pageSize });
    } else {
      pdf.addPage(pageSize === 'fit' ? [pW, pH] : pageSize, pW > pH ? 'landscape' : 'portrait');
    }
    const pageAspect = pW / pH;
    let drawW, drawH, drawX, drawY;
    if (imgAspect > pageAspect) { drawW = pW; drawH = pW / imgAspect; drawX = 0; drawY = (pH - drawH) / 2; }
    else { drawH = pH; drawW = pH * imgAspect; drawX = (pW - drawW) / 2; drawY = 0; }
    pdf.addImage(processed, 'JPEG', drawX, drawY, drawW, drawH);
  }
  pdf.save(filename + '.pdf');
}

function processPageForExport(page) {
  return new Promise((resolve) => {
    const img = new Image();
    img.onload = () => {
      const c = document.createElement('canvas');
      c.width = img.width; c.height = img.height;
      const ctx = c.getContext('2d');
      const b = 100 + page.adjustments.brightness;
      const cont = 100 + page.adjustments.contrast;
      let f = `brightness(${b}%) contrast(${cont}%)`;
      if (page.adjustments.bw) f += ' grayscale(100%)';
      ctx.filter = f; ctx.drawImage(img, 0, 0);
      resolve(c.toDataURL('image/jpeg', 0.92));
    };
    img.src = page.imageData;
  });
}

function loadImage(src) {
  return new Promise(r => { const i = new Image(); i.onload = () => r(i); i.src = src; });
}

## 5. UPDATED DETAIL RENDERER (replace the `openDetail` function's content builder)

Replace the existing `updatedHtml` line:

```javascript
  var updatedHtml = item.updated ? '<p class="detail-updated">Last reviewed: ' + item.updated + '</p>' : '';
```

With:

```javascript
  var updatedHtml = '';
  if (item.updated || (item.regions && item.regions.length)) {
    updatedHtml = '<p class="detail-updated">';
    if (item.updated) updatedHtml += 'Last reviewed: ' + item.updated;
    if (item.regions && item.regions.length && item.regions[0] !== 'global') {
      var regionLabels = { uk: 'UK', us: 'US', eu: 'EU', global: 'Global' };
      updatedHtml += (item.updated ? ' \u00B7 ' : '') + item.regions.map(function(r) { return '<span class="region-badge">' + (regionLabels[r] || r) + '</span>'; }).join('');
    }
    updatedHtml += '</p>';
```

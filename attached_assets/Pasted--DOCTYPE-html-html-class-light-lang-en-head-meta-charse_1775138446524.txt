function getTheme() {
  return localStorage.getItem('kin039-theme') || 'light';

function applyTheme(t) {
  document.documentElement.className = t;
  var meta = document.getElementById('theme-color-meta');
  meta.content = t === 'dark' ? '#121010' : '#F5F1EB';
  document.getElementById('themeToggle').textContent = t === 'dark' ? '\u2600\uFE0F' : '\uD83C\uDF19';
  localStorage.setItem('kin039-theme', t);
}

document.getElementById('themeToggle').addEventListener('click', function() {
  applyTheme(getTheme() === 'dark' ? 'light' : 'dark');
});

  searchInput.focus();

// ── CARDS ──
function getFilteredData() {
  var collectionIds = null;
  if (activeCollection) {
    var col = COLLECTIONS.find(function(c) { return c.id === activeCollection; });
    if (col) collectionIds = col.entries;
  }
  return DATA.filter(function(item) {
    var matchCollection = !collectionIds || collectionIds.indexOf(item.id) !== -1;
    var matchFilter = activeFilter === 'all' || item.tags.indexOf(activeFilter) !== -1;
    var matchSearch = !searchQuery || 
      item.title.toLowerCase().indexOf(searchQuery) !== -1 ||
      item.hook.toLowerCase().indexOf(searchQuery) !== -1 ||
      item.real.toLowerCase().indexOf(searchQuery) !== -1 ||
      item.fake.toLowerCase().indexOf(searchQuery) !== -1 ||
      item.bigPicture.toLowerCase().indexOf(searchQuery) !== -1 ||
      item.trustMarkers.toLowerCase().indexOf(searchQuery) !== -1 ||
      (item.healthNote && item.healthNote.toLowerCase().indexOf(searchQuery) !== -1) ||
      (item.envNote && item.envNote.toLowerCase().indexOf(searchQuery) !== -1) ||
      item.howToTell.some(function(tip) { return tip.toLowerCase().indexOf(searchQuery) !== -1; }) ||
      item.tags.some(function(t) { return t.indexOf(searchQuery) !== -1; });
    return matchCollection && matchFilter && matchSearch;
  });
}

function renderCards() {
  var container = document.getElementById('cards');
  var filtered = getFilteredData();
  container.innerHTML = '';

  document.getElementById('entryCount').textContent = filtered.length + ' of ' + DATA.length + ' entries';

  if (filtered.length === 0) {
    container.innerHTML = '<div class="no-results"><h3>Nothing found</h3><p>Try a different search or filter.</p></div>';
    return;
  }

  filtered.forEach(function(item) {
    var card = document.createElement('div');
    card.className = 'card';
    card.setAttribute('role', 'button');
    card.setAttribute('tabindex', '0');

    var tagsHtml = item.tags.map(function(t) {
      return '<span class="card-tag ' + t + '">' + t + '</span>';
    }).join('');

    card.innerHTML = '<div class="card-top"><span class="card-icon">' + item.icon + '</span><div class="card-tags">' + tagsHtml + '</div></div>' +
      '<h2>' + item.title + '</h2>' +
      '<p class="hook">' + item.hook + '</p>' +
      '<p class="tap-hint">Tap to read more \u2192</p>';

    card.addEventListener('click', function() { openDetail(item); });
    card.addEventListener('keydown', function(e) {
      if (e.key === 'Enter' || e.key === ' ') { e.preventDefault(); openDetail(item); }
    });

    container.appendChild(card);
  });
}

// ── DETAIL ──
function openDetail(item) {
  var overlay = document.getElementById('detailOverlay');
  var content = document.getElementById('detailContent');

  var tagsHtml = item.tags.map(function(t) {
    return '<span class="card-tag ' + t + '">' + t + '</span>';
  }).join('');

  var howToTellHtml = item.howToTell.map(function(tip) {
    return '<li>' + tip + '</li>';
  }).join('');

  var healthHtml = item.healthNote ? '<div class="info-box health"><strong>\u26A0 Health</strong>' + item.healthNote + '</div>' : '';
  var envHtml = item.envNote ? '<div class="info-box env"><strong>\uD83C\uDF0D Environment</strong>' + item.envNote + '</div>' : '';

  var sourcesHtml = '';
  if (item.sources && item.sources.length > 0) {
    sourcesHtml = '<div class="detail-section"><h3>Sources</h3><div class="detail-sources">' +
      item.sources.map(function(s) { return '<a href="' + s.url + '" target="_blank" rel="noopener">' + s.label + ' \u2197</a>'; }).join('') +
      '</div></div>';
  }

  var updatedHtml = item.updated ? '<p class="detail-updated">Last reviewed: ' + item.updated + '</p>' : '';

  content.innerHTML = '<div class="detail-icon">' + item.icon + '</div>' +
    '<div class="detail-tags">' + tagsHtml + '</div>' +
    '<h1>' + item.title + '</h1>' +
    '<p class="detail-hook">' + item.hook + '</p>' +
    updatedHtml +

    (item.inShort ? '<div class="in-short">\u{1F4CC} ' + item.inShort + '</div>' : '') +

    '<div class="detail-section"><h3>The Real Thing</h3>' +
    '<div class="info-box real">' + item.real + '</div></div>' +

    '<div class="detail-section"><h3>What You Are Actually Getting</h3>' +
    '<div class="info-box fake">' + item.fake + '</div></div>' +

    '<div class="detail-section"><h3>How to Tell</h3><ul>' + howToTellHtml + '</ul></div>' +

    '<div class="detail-section"><h3>The Bigger Picture</h3><p>' + item.bigPicture + '</p></div>' +

    (healthHtml || envHtml ? '<div class="detail-section"><h3>Impact</h3>' + healthHtml + envHtml + '</div>' : '') +

    '<div class="detail-section"><h3>What to Trust</h3><p>' + item.trustMarkers + '</p></div>' +

    sourcesHtml +

    '<button class="detail-share" onclick="shareEntry(\'' + item.id + '\', \'' + item.title.replace(/'/g, "\\'") + '\')">\u{1F517} Share this entry</button>';

  overlay.classList.add('open');
  overlay.scrollTop = 0;
  document.body.style.overflow = 'hidden';
  history.pushState({ detail: true }, '', '#' + item.id);
}

function closeDetail() {
  document.getElementById('detailOverlay').classList.remove('open');
  document.body.style.overflow = '';
  if (window.location.hash) history.pushState('', '', window.location.pathname);
}

document.getElementById('detailClose').addEventListener('click', closeDetail);
document.getElementById('detailOverlay').addEventListener('click', function(e) {
  if (e.target === this) closeDetail();
});

  };
  fetch('https://script.google.com/macros/s/AKfycbxBRGfOmtQUxyaBGjYVj2mtKinI7qlGm1v921K49TiBDP5RUY9CWK_M-vpLCm2HWJxhuA/exec', {

var activeCollection = null;

function renderCollections() {
  var el = document.getElementById('collections');
  if (!COLLECTIONS || COLLECTIONS.length === 0) { el.innerHTML = ''; return; }
  var label = '<p class="collections-label">Browse by situation</p>';
  var chips = COLLECTIONS.map(function(col) {
    return '<button class="collection-chip' + (activeCollection === col.id ? ' active' : '') + '" data-col="' + col.id + '">' +
      '<div class="collection-chip-title">' + col.title + '</div>' +
      '<div class="collection-chip-desc">' + col.desc + '</div>' +
      '</button>';
  }).join('');
  el.innerHTML = label + '<div class="collections-row">' + chips + '</div>';
  el.querySelectorAll('.collection-chip').forEach(function(btn) {
    btn.addEventListener('click', function() {
      var colId = this.getAttribute('data-col');
      if (activeCollection === colId) {
        activeCollection = null;
      } else {
        activeCollection = colId;
        activeFilter = 'all';
      }
      renderCollections();
      renderFilters();
      renderCards();
    });
  });
}

function shareEntry(id, title) {
  var url = window.location.origin + window.location.pathname + '#' + id;
  if (navigator.share) {
    navigator.share({ title: 'Is This Real? - ' + title, url: url }).catch(function() {});
  } else if (navigator.clipboard) {
    navigator.clipboard.writeText(url).then(function() {
      showToast('Link copied');
    }).catch(function() {
      showToast('Could not copy link');
    });
  } else {
    showToast('Share not supported');
  }
}

  var mt=act.reduce(function(a,s){return a+normM(s);},0),ps=inact.reduce(function(a,s){return a+normM(s);},0);
  var h='<div class="card"><div class="card-title">Subscriptions</div><div class="stat-grid">'+sbox(fmtI(mt),'Monthly')+sbox(fmtI(Math.round(mt*12)),'Yearly')+'</div>';
  if(ps>0)h+='<div class="result-box">'+rrow('Potential savings (inactive)',fmt(ps)+'/mo')+'</div>';
  h+='</div><div class="card"><div class="card-title">Add Subscription</div><input type="text" id="sn" placeholder="Name"><div style="display:flex;gap:8px"><input type="number" inputmode="decimal" id="sco" placeholder="Cost" style="flex:1"><select id="scy" style="flex:1"><option value="monthly">Monthly</option><option value="weekly">Weekly</option><option value="quarterly">Quarterly</option><option value="yearly">Yearly</option></select></div><button class="btn-primary" id="s-add">Add</button></div>';
  function sl(arr,t){if(!arr.length)return'';var o='<div class="card"><div class="card-title">'+t+'</div>';arr.forEach(function(s){o+='<div class="list-item"><div><div class="li-name">'+esc(s.name)+'</div><div class="li-detail">'+fmt(s.cost)+'/'+s.billingCycle+' ('+fmt(normM(s))+'/mo)</div></div><div class="li-actions"><button class="icon-btn" data-ts="'+s.id+'">'+(s.active?'⏸':'▶')+'</button><button class="icon-btn" data-rs="'+s.id+'">✕</button></div></div>';});return o+'</div>';}
  h+=sl(act,'Active')+sl(inact,'Inactive');
  if(!subs.length)h+='<div class="card"><div class="empty">No subscriptions tracked yet.</div></div>';
  viewEl.innerHTML=h;
  document.getElementById('s-add').onclick=function(){var n=document.getElementById('sn').value.trim(),c=parseFloat(document.getElementById('sco').value),cy=document.getElementById('scy').value;if(!n||isNaN(c))return;S.subscriptions.push({id:uid(),name:n,cost:c,billingCycle:cy,category:'',active:true,addedAt:lnow()});save();rSubs();};
  document.querySelectorAll('[data-ts]').forEach(function(b){b.onclick=function(){var s=S.subscriptions.find(function(x){return x.id===b.dataset.ts;});if(s){s.active=!s.active;save();rSubs();}};});
  document.querySelectorAll('[data-rs]').forEach(function(b){b.onclick=function(){S.subscriptions=S.subscriptions.filter(function(x){return x.id!==b.dataset.rs;});save();rSubs();};});
}

// DELAY
function rDelay(){
  var items=S.impulseItems||[],now=Date.now(),ch=false;
  items.forEach(function(it){if(it.status==='pending'){var cr=new Date(it.createdAt).getTime(),dl=(it.delayHours||72)*36e5;if(now-cr>=dl){it.status='expired';it.resolvedAt=lnow();ch=true;}}});
  if(ch)save();
  var pend=items.filter(function(i){return i.status==='pending';}),res=items.filter(function(i){return i.status!=='pending';});
  var sv=res.filter(function(i){return i.status==='skipped'||i.status==='expired';}).reduce(function(a,i){return a+(i.price||0);},0);
  var h='<div class="card"><div class="card-title">72-Hour Rule</div><input type="text" id="dn" placeholder="What do you want to buy?"><input type="number" inputmode="decimal" id="dp" placeholder="Price"><button class="btn-primary" id="d-add">Start Timer</button></div>';
  if(sv>0)h+='<div class="card"><div class="stat-box" style="margin:0"><div class="stat-val" style="color:var(--accent)">'+fmtI(sv)+'</div><div class="stat-lbl">Money not spent</div></div></div>';
  h+='<div class="card"><div class="card-title">Pending</div>';
  if(!pend.length)h+='<div class="empty">No items waiting.</div>';
  pend.forEach(function(it){var cr=new Date(it.createdAt).getTime(),dl=(it.delayHours||72)*36e5,rm=Math.max(0,dl-(now-cr));h+='<div class="list-item" style="flex-wrap:wrap"><div style="flex:1"><div class="li-name">'+esc(it.name)+' <span style="font-weight:400;color:var(--text-light)">'+fmt(it.price)+'</span></div><div class="countdown" data-tid="'+it.id+'" data-rem="'+rm+'">'+fmtMs(rm)+'</div></div><div class="li-actions" style="margin-top:4px"><button class="icon-btn" data-buy="'+it.id+'" title="Bought it">💸</button><button class="icon-btn" data-skip="'+it.id+'" title="Skipped">✕</button></div></div>';});
  h+='</div>';
  if(res.length){h+='<div class="card"><div class="card-title">History</div>';res.slice(-20).reverse().forEach(function(it){h+='<div class="list-item"><div><div class="li-name">'+esc(it.name)+' <span style="font-weight:400;color:var(--text-light)">'+fmt(it.price)+'</span></div></div><span class="tag '+(it.status==='bought'?'tag-red':'tag-green')+'">'+it.status+'</span></div>';});h+='</div>';}
  viewEl.innerHTML=h;
  if(pend.length){dTimer=setInterval(function(){var els=document.querySelectorAll('[data-tid]'),exp=false;els.forEach(function(el){var r=parseInt(el.dataset.rem)-1000;if(r<0)r=0;el.dataset.rem=r;el.textContent=fmtMs(r);if(r===0)exp=true;});if(exp){clearInterval(dTimer);rDelay();}},1000);}
  document.getElementById('d-add').onclick=function(){var n=document.getElementById('dn').value.trim(),p=parseFloat(document.getElementById('dp').value);if(!n||isNaN(p))return;S.impulseItems.push({id:uid(),name:n,price:p,category:'',createdAt:lnow(),delayHours:72,status:'pending',resolvedAt:null});save();rDelay();};
  document.querySelectorAll('[data-buy]').forEach(function(b){b.onclick=function(){var it=S.impulseItems.find(function(i){return i.id===b.dataset.buy;});if(it){it.status='bought';it.resolvedAt=lnow();save();rDelay();}};});
  document.querySelectorAll('[data-skip]').forEach(function(b){b.onclick=function(){var it=S.impulseItems.find(function(i){return i.id===b.dataset.skip;});if(it){it.status='skipped';it.resolvedAt=lnow();save();rDelay();}};});
}

// BUFFER
function rBuffer(){
  var sv=S.savings,ec=(S.costs||[]).filter(function(c){return c.essential;}).reduce(function(a,c){return a+(c.amount||0);},0);
  var hist=S.bufferHistory||[],al=S.config.bufferAlertMonths||2;
  var h='<div class="card"><div class="card-title">Emergency Buffer</div><label>Total savings</label><input type="number" inputmode="decimal" id="bs" value="'+(sv!=null?sv:'')+'" placeholder="0"><label>Monthly essential spend</label><input type="number" inputmode="decimal" id="be" value="'+(ec||'')+'" placeholder="0"><button class="btn-primary" id="b-calc">Update</button><div id="b-out"></div></div>';
  if(hist.length>=2){h+='<div class="card"><div class="card-title">Trend</div>';hist.slice(-6).forEach(function(e,i,arr){var ar='';if(i>0){var d=e.savings-arr[i-1].savings;ar=d>0?' ↑':d<0?' ↓':' →';}h+='<div class="list-item"><div class="li-detail">'+e.recordedAt.slice(0,10)+'</div><div class="result-value">'+fmtI(e.savings)+ar+'</div></div>';});h+='</div>';}
  viewEl.innerHTML=h;
  function calc(){var s=parseFloat(document.getElementById('bs').value),e=parseFloat(document.getElementById('be').value);if(isNaN(s))return;S.savings=s;var last=hist[hist.length-1];if(!last||last.savings!==s){S.bufferHistory.push({savings:s,recordedAt:lnow()});if(S.bufferHistory.length>12)S.bufferHistory.shift();}save();if(isNaN(e)||e<=0){document.getElementById('b-out').innerHTML='<div class="result-box"><p style="font-size:13px;color:var(--text-light)">Add essential costs in Budget or enter monthly essentials above.</p></div>';return;}var m=s/e,cls=m<al?'danger-box':m<al*2?'warn-box':'result-box';var o='<div class="'+cls+'">'+rrow('Months remaining',m.toFixed(1));if(m<al)o+='<p style="font-size:13px;font-weight:600;color:var(--danger);margin-top:8px">Below your '+al+'-month alert threshold.</p>';o+='</div>';document.getElementById('b-out').innerHTML=o;}
  document.getElementById('b-calc').onclick=calc;if(sv!=null&&ec>0)calc();
}

// PULSE
function rPulse(){
  var inc=S.config.income,costs=S.costs||[],subs=S.subscriptions||[],items=S.impulseItems||[],sv=S.savings,hist=S.bufferHistory||[];
  var ec=costs.filter(function(c){return c.essential;}).reduce(function(a,c){return a+(c.amount||0);},0);
  var ct=costs.reduce(function(a,c){return a+(c.amount||0);},0);
  var st=subs.filter(function(s){return s.active;}).reduce(function(a,s){return a+normM(s);},0);
  var rem=inc?inc-ct-st:null,now=new Date(),dim=new Date(now.getFullYear(),now.getMonth()+1,0).getDate(),dl=dim-now.getDate()+1,ds=rem!=null?rem/dl:null;
  var pend=items.filter(function(i){return i.status==='pending';}),ms=items.filter(function(i){return i.status==='skipped'||i.status==='expired';}).reduce(function(a,i){return a+(i.price||0);},0);
  var wa=Date.now()-7*864e5,rr=items.filter(function(i){return i.status!=='pending'&&i.resolvedAt&&new Date(i.resolvedAt).getTime()>wa;});
  var months=sv&&ec>0?sv/ec:null;var lt=hist.slice(-2),trend='';if(lt.length===2){var d=lt[1].savings-lt[0].savings;trend=d>0?'↑ Growing':d<0?'↓ Shrinking':'→ Stable';}
  function sec(t,has,c,nav,last){return'<div class="pulse-section"'+(last?' style="border-bottom:none"':'')+'><div class="pulse-title">'+t+'</div>'+(has?c:'<div class="pulse-empty" data-nav="'+nav+'">No data yet. Tap to configure.</div>')+'</div>';}
  var h='<div class="card"><div class="card-title">Weekly Pulse</div>';
  h+=sec('Budget',!!inc,'<div class="pulse-val">Daily safe spend: '+fmtI(ds)+' ('+dl+' days left)</div>','settings');
  h+=sec('Subscriptions',subs.length>0,'<div class="pulse-val">'+fmtI(st)+'/mo across '+subs.filter(function(s){return s.active;}).length+' active</div>'+(subs.filter(function(s){return!s.active;}).length>0?'<div style="font-size:13px;color:var(--warn)">'+subs.filter(function(s){return!s.active;}).length+' inactive flagged</div>':''),'subs');
  h+=sec('72-Hour Rule',items.length>0,'<div class="pulse-val">'+pend.length+' pending, '+rr.length+' resolved this week</div>'+(ms>0?'<div style="font-size:13px;color:var(--accent);font-weight:600">Total saved: '+fmtI(ms)+'</div>':''),'delay');
  h+=sec('Buffer',months!=null,'<div class="pulse-val">'+months.toFixed(1)+' months runway '+(trend?'<span style="font-size:13px;color:var(--text-light)">'+trend+'</span>':'')+'</div>','buffer',true);
  h+='</div>';viewEl.innerHTML=h;
  document.querySelectorAll('[data-nav]').forEach(function(el){el.onclick=function(){goTab(el.dataset.nav);};});
}

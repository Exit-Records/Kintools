var renderQueue = [];
var rendering = false;

function queueRender(origIndex, rotation, canvas) {
  renderQueue.push({ origIndex: origIndex, rotation: rotation, canvas: canvas });
  processQueue();
}

async function processQueue() {
  if (rendering || renderQueue.length === 0) return;
  rendering = true;
  var job = renderQueue.shift();
  try {
    var page = await pdfDoc.getPage(job.origIndex + 1);
    var baseVp = page.getViewport({ scale: 1 });
    var renderSize = ZOOM_RENDER_SIZE[zoomIdx] * (window.devicePixelRatio || 1);
    var scale = renderSize / Math.max(baseVp.width, baseVp.height);
    var viewport = page.getViewport({ scale: scale, rotation: job.rotation });
    job.canvas.width = viewport.width;
    job.canvas.height = viewport.height;
    var ctx = job.canvas.getContext('2d');
    await page.render({ canvasContext: ctx, viewport: viewport }).promise;
    var loadingEl = job.canvas.parentElement.querySelector('.loading');
    if (loadingEl) loadingEl.style.display = 'none';
  } catch (e) {
    console.error('Render error:', e);
  }
  rendering = false;
  processQueue();
}

function parseRange(str, max) {
  var result = [];
  str.split(',').forEach(function(part) {
    part = part.trim();
    var m = part.match(/^(\d+)\s*-\s*(\d+)$/);
    if (m) {
      var a = parseInt(m[1]) - 1, b = parseInt(m[2]) - 1;
      if (a > b) { var tmp = a; a = b; b = tmp; }
      for (var i = Math.max(0, a); i <= Math.min(max - 1, b); i++) result.push(i);
    } else {
      var n = parseInt(part) - 1;
      if (!isNaN(n) && n >= 0 && n < max) result.push(n);
    }
  });
  return Array.from(new Set(result)).sort(function(a, b) { return a - b; });
}

window.exportPdf = async function(mode) {
  var toExport;
  if (mode === 'selected') {
    toExport = pages.filter(function(p) { return p.selected; });
  } else if (mode === 'all') {
    toExport = pages;
  } else if (mode === 'split') {
    toExport = pages.filter(function(p) { return p.selected; });
    if (toExport.length === 0) toExport = pages;
  }

  if (!toExport || toExport.length === 0) { showToast('No pages to export'); return; }

  showToast('Building PDF...');

  try {
    var lib = await loadPdfLib();
    var PDFDocument = lib.PDFDocument;
    var degrees = lib.degrees;
    var srcDoc = await PDFDocument.load(pdfBytes);

    if (mode === 'split') {
      for (var i = 0; i < toExport.length; i++) {
        var singleDoc = await PDFDocument.create();
        var copied = await singleDoc.copyPages(srcDoc, [toExport[i].origIndex]);
        var pg = singleDoc.addPage(copied[0]);
        if (toExport[i].rotation !== 0) pg.setRotation(degrees(toExport[i].rotation));
        var outBytes = await singleDoc.save();
        downloadBlob(outBytes, baseName() + '-page' + (toExport[i].origIndex + 1) + '.pdf');
        if (i < toExport.length - 1) await sleep(300);
      }
      showToast('Exported ' + toExport.length + ' separate PDFs');
    } else {
      var newDoc = await PDFDocument.create();
      var indices = toExport.map(function(p) { return p.origIndex; });
      var copiedPages = await newDoc.copyPages(srcDoc, indices);
      toExport.forEach(function(p, i) {
        var pg = newDoc.addPage(copiedPages[i]);
        if (p.rotation !== 0) pg.setRotation(degrees(p.rotation));
      });
      var outBytes = await newDoc.save();
      var suffix = mode === 'selected' ? '-selected' : '-export';
      downloadBlob(outBytes, baseName() + suffix + '.pdf');
      showToast('Exported ' + toExport.length + ' pages');
    }
  } catch (err) {
    console.error(err);
    showToast('Export failed: ' + err.message);
  }
};

  var blob = new Blob([bytes], { type: 'application/pdf' });

}

document.addEventListener('keydown', function(e) {
  if (e.target.tagName === 'INPUT' || e.target.tagName === 'TEXTAREA' || e.target.tagName === 'SELECT') return;
  if (!pdfDoc || pages.length === 0) return;

  var key = e.key;

  if ((e.ctrlKey || e.metaKey) && key === 'z' && !e.shiftKey) { e.preventDefault(); window.undo(); return; }
  if ((e.ctrlKey || e.metaKey) && (key === 'Z' || (key === 'z' && e.shiftKey))) { e.preventDefault(); window.redo(); return; }

  if (key === 'a' && !e.ctrlKey && !e.metaKey) { e.preventDefault(); window.selectAll(); return; }
  if (key === 'd' && !e.ctrlKey && !e.metaKey) { e.preventDefault(); window.selectNone(); return; }
  if (key === 'i' && !e.ctrlKey && !e.metaKey) { e.preventDefault(); window.invertSelection(); return; }

  if (key === 'Delete' || key === 'Backspace') { e.preventDefault(); window.removeSelected(); return; }

  if (key === 'r' && !e.ctrlKey && !e.metaKey && !e.shiftKey) {
    e.preventDefault();
    var sel = pages.filter(function(p) { return p.selected; });
    if (sel.length > 0) window.rotateSelected(90);
    else if (focusIdx >= 0) window.rotatePage(focusIdx, 90);
    return;
  }
  if (key === 'R' && !e.ctrlKey && !e.metaKey) {
    e.preventDefault();
    var sel = pages.filter(function(p) { return p.selected; });
    if (sel.length > 0) window.rotateSelected(-90);
    else if (focusIdx >= 0) window.rotatePage(focusIdx, -90);
    return;
  }

  if (key === 'ArrowRight' || key === 'ArrowDown') {
    e.preventDefault();
    focusIdx = Math.min(pages.length - 1, focusIdx + 1);
    var fc = pageGrid.children[focusIdx];
    if (fc) fc.focus();
    return;
  }
  if (key === 'ArrowLeft' || key === 'ArrowUp') {
    e.preventDefault();
    focusIdx = Math.max(0, focusIdx - 1);
    var fc = pageGrid.children[focusIdx];
    if (fc) fc.focus();
    return;
  }

  if (key === ' ') {
    e.preventDefault();
    if (focusIdx >= 0 && focusIdx < pages.length) {
      saveState('toggle select');
      pages[focusIdx].selected = !pages[focusIdx].selected;
      lastClickIdx = focusIdx;
      rebuildGrid();
    }
    return;
  }
});

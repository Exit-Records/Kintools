// ── Colour math ──
function hexToRgb(hex) {
  hex = hex.replace('#','').trim();
  if (hex.length === 3) hex = hex[0]+hex[0]+hex[1]+hex[1]+hex[2]+hex[2];
  return [parseInt(hex.slice(0,2),16), parseInt(hex.slice(2,4),16), parseInt(hex.slice(4,6),16)];
}
function rgbToHex(r,g,b) {
  return '#' + [r,g,b].map(v => Math.round(Math.max(0,Math.min(255,v))).toString(16).padStart(2,'0')).join('');
}
function rgbToHsl(r,g,b) {
  r/=255; g/=255; b/=255;
  const max=Math.max(r,g,b), min=Math.min(r,g,b), l=(max+min)/2;
  if(max===min) return [0,0,Math.round(l*100)];
  const d=max-min, s=l>0.5?d/(2-max-min):d/(max+min);
  let h=0;
  if(max===r) h=(g-b)/d+(g<b?6:0);
  else if(max===g) h=(b-r)/d+2;
  else h=(r-g)/d+4;
  return [Math.round(h/6*360), Math.round(s*100), Math.round(l*100)];
}
function hslToRgb(h,s,l) {
  s/=100; l/=100;
  const a=s*Math.min(l,1-l);
  const f=n=>{const k=(n+h/30)%12; return l-a*Math.max(Math.min(k-3,9-k,1),-1);};
  return [Math.round(f(0)*255), Math.round(f(8)*255), Math.round(f(4)*255)];
}
function hslToHex(h,s,l) { const [r,g,b]=hslToRgb(h,s,l); return rgbToHex(r,g,b); }
function clamp(v,lo,hi) { return Math.max(lo,Math.min(hi,v)); }
function luminance(r,g,b) {
  const [rl,gl,bl] = [r,g,b].map(v => { v/=255; return v<=0.03928?v/12.92:Math.pow((v+0.055)/1.055,2.4); });
  return 0.2126*rl + 0.7152*gl + 0.0722*bl;
}
function contrastRatio(r1,g1,b1,r2,g2,b2) {
  const l1=luminance(r1,g1,b1), l2=luminance(r2,g2,b2);
  return (Math.max(l1,l2)+0.05)/(Math.min(l1,l2)+0.05);
}
function textColour(r,g,b) { return luminance(r,g,b)>0.179?'#1a1a1a':'#f5f5f5'; }
function colourDescription(h,s,l) {
  if(s<8) return l<12?'Near Black':l<30?'Dark Grey':l<55?'Grey':l<80?'Light Grey':'Near White';
  if(l<15) return 'Very Dark';
  if(l>90) return 'Very Light';
  const prefix = s<30?'Muted ':s>75?'Vivid ':'';
  const shade = l<35?'Dark ':l>65?'Light ':'';
  let name='';
  if(h<15||h>=345) name='Red';
  else if(h<40) name='Orange';
  else if(h<65) name='Yellow';
  else if(h<80) name='Yellow-Green';
  else if(h<150) name='Green';
  else if(h<175) name='Teal';
  else if(h<210) name='Cyan';
  else if(h<255) name='Blue';
  else if(h<285) name='Indigo';
  else if(h<315) name='Violet';
  else name='Pink';
  return prefix+shade+name;
}

// ── Harmony ──
function getHarmony(h,s,l,mode,temp) {
  const rot = temp*15;
  const shift = n => (h+n+rot+360)%360;
  if(mode==='complementary') return [hslToHex(shift(0),s,l), hslToHex(shift(180),s,l)];
  if(mode==='analogous') return [hslToHex(shift(-30),s,l), hslToHex(shift(0),s,l), hslToHex(shift(30),s,l)];
  return [hslToHex(shift(0),s,l), hslToHex(shift(120),s,l), hslToHex(shift(240),s,l)];
}
function getShades(h,s) {
  return [5,12,20,30,40,50,60,70,80,90,95].map(l => hslToHex(h,s,l));
}

// ── Wheel drawing ──
// Matches original React implementation: atan2 + 180 so red sits at 9 o'clock
const WS = 220, WR = WS/2-4, WCX = WS/2, WCY = WS/2;
function drawWheel() {
  const ctx = wheelCanvas.getContext('2d');
  const img = ctx.createImageData(WS, WS);
  for(let y=0; y<WS; y++) {
    for(let x=0; x<WS; x++) {
      const dx=x-WCX, dy=y-WCY, dist=Math.sqrt(dx*dx+dy*dy);
      const i = (y*WS+x)*4;
      if(dist>WR) { img.data[i+3]=0; continue; }
      const angle = Math.atan2(dy,dx)*180/Math.PI + 180;
      const s2 = dist/WR*100;
      const [r,g,b] = hslToRgb(angle, s2, 50);
      img.data[i]=r; img.data[i+1]=g; img.data[i+2]=b; img.data[i+3]=255;
    }
  }
  ctx.putImageData(img, 0, 0);
}
drawWheel();

function updateWheelCursor() {
  // Convert hue back to canvas position: inverse of the +180 used when drawing
  const angle = hue*Math.PI/180 - Math.PI;
  const dist = sat/100*WR;
  const cx = WCX + Math.cos(angle)*dist;
  const cy = WCY + Math.sin(angle)*dist;
  wheelCursor.style.left = (cx/WS*100)+'%';
  wheelCursor.style.top = (cy/WS*100)+'%';
  wheelCursor.style.background = hslToHex(hue, sat, 50);
}

function wheelPointer(e) {
  if(!(e.buttons & 1) && e.type !== 'pointerdown') return;
  const rect = wheelCanvas.getBoundingClientRect();
  const scaleX = WS/rect.width, scaleY = WS/rect.height;
  const dx = (e.clientX - rect.left)*scaleX - WCX;
  const dy = (e.clientY - rect.top)*scaleY - WCY;
  const dist = Math.min(Math.sqrt(dx*dx+dy*dy), WR);
  // Raw angle: cursor renders opposite via the +180 offset in drawing/positioning
  const angle = (Math.atan2(dy, dx)*180/Math.PI + 360) % 360;
  const s = Math.round(dist/WR*100);
  setColour(Math.round(angle), s, lit);
}
wheelCanvas.addEventListener('pointerdown', e => { wheelPointer(e); wheelCanvas.setPointerCapture(e.pointerId); });
wheelCanvas.addEventListener('pointermove', wheelPointer);

}

function render() {
  const hex = hslToHex(hue, sat, lit);
  const [r,g,b] = hexToRgb(hex);

  const shades = getShades(hue, sat);
  const whiteRatio = contrastRatio(r,g,b,255,255,255);
  const blackRatio = contrastRatio(r,g,b,0,0,0);
  const bestRatio = Math.max(whiteRatio, blackRatio);

  // CVD filter
  cvdWrap.style.filter = cvdMode !== 'none' ? `url(#cvd-${cvdMode})` : '';

  // Header
  colourHeader.style.background = hex;
  colourHeader.style.color = txtCol;

  // Previews

  previewOnBg.style.color = hex;

  // Gradient bar

  // CVD buttons
  renderCvdButtons(txtCol, hex);

  // Action button styles
  document.querySelectorAll('.action-btn').forEach(btn => { btn.style.color = txtCol; });
  statusMsg.style.color = txtCol;

  // Inputs
  inputHex.value = hex;
  inputRgb.value = `${r}, ${g}, ${b}`;
  inputHsl.value = `${hue}, ${sat}%, ${lit}%`;

  // Wheel cursor
  updateWheelCursor();

  // Sliders
  sliderLightness.value = lit;
  lightnessVal.textContent = lit + '%';
  sliderSaturation.value = sat;
  saturationVal.textContent = sat + '%';
  sliderRotate.value = hueRotate;
  rotateVal.textContent = hueRotate + '°';
  sliderTemp.value = tempShift;

  renderHarmonyModes();

  renderSwatches(palette, hex);

function makeSwatch(hex, size, opts = {}) {
  const div = document.createElement('div');
  div.className = 'swatch' + (opts.active ? ' active-swatch' : '');
  div.style.width = size + 'px';
  div.style.height = size + 'px';
  div.style.background = hex;
  div.style.border = opts.active ? '3px solid white' : '2px solid rgba(0,0,0,0.08)';
  const [cr,cg,cb] = hexToRgb(hex);
  const tc = textColour(cr, cg, cb);

  // Copy button overlay
  const copyBtn = document.createElement('button');
  copyBtn.className = 'copy-overlay';
  copyBtn.style.color = tc;
  copyBtn.textContent = 'copy';
  copyBtn.addEventListener('click', e => {
    e.stopPropagation();
    navigator.clipboard.writeText(hex).then(() => {
      copyBtn.textContent = '✓';
      setTimeout(() => { copyBtn.textContent = 'copy'; }, 1200);
    });
  });
  div.appendChild(copyBtn);

  if(opts.onClick) div.addEventListener('click', opts.onClick);

  // Lock button
  if(opts.onToggleLock !== undefined) {
    const lockBtn = document.createElement('button');
    lockBtn.className = 'lock-btn';
    lockBtn.textContent = opts.locked ? '🔒' : '○';
    lockBtn.title = opts.locked ? 'Unlock' : 'Lock';
    lockBtn.addEventListener('click', e => { e.stopPropagation(); opts.onToggleLock(); });
    div.appendChild(lockBtn);
  }

  return div;
}

function renderSwatches(palette, currentHex) {
  harmonySwatches.innerHTML = '';

      active: hex.toLowerCase() === currentHex.toLowerCase(),
      onClick: () => setColourFromHex(hex),
      locked: lockedIndices.has(i),
      onToggleLock: () => {
        if(lockedIndices.has(i)) lockedIndices.delete(i);
        else lockedIndices.add(i);
        render();
      }
    });
    col.appendChild(sw);
    const label = document.createElement('div');
    label.className = 'swatch-label';
    label.textContent = hex;
    col.appendChild(label);
    harmonySwatches.appendChild(col);
  });
}

function renderRecent(currentHex) {

// Shuffle
function shufflePalette() {
  const palette = getHarmony((hue+hueRotate)%360, sat, lit, harmonyMode, tempShift);
  palette.forEach((_, i) => {
    if(!lockedIndices.has(i)) {
      if(i===0) hue = Math.round(Math.random()*360);
    }
  });
  hueRotate = Math.round(Math.random()*360);
  render();
}
$('btn-shuffle').addEventListener('click', shufflePalette);

// Spacebar shuffle
window.addEventListener('keydown', e => {
  if(e.code === 'Space' && !(e.target instanceof HTMLInputElement) && !(e.target instanceof HTMLTextAreaElement)) {
    e.preventDefault();
    shufflePalette();
  }
});

// ── URL hash load ──
function loadFromHash() {
  const hash = location.hash.replace('#','');
  if(hash && /^[0-9a-fA-F]{6}(-[0-9a-fA-F]{6})*$/.test(hash)) {
    try {
      const first = '#' + hash.split('-')[0];
      setColourFromHex(first);
    } catch(e){}
  }
}

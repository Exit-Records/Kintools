  // ==================== CRYPTO RANDOM ====================
  function secureRandom(max) {
    const arr = new Uint32Array(1);
    crypto.getRandomValues(arr);
    return arr[0] % max;
  }

  function secureRandomFloat() {
    const arr = new Uint32Array(1);
    crypto.getRandomValues(arr);
    return arr[0] / (0xFFFFFFFF + 1);
  }

  // ==================== CHARSETS ====================
  const CHARSETS = {
    upper: 'ABCDEFGHIJKLMNOPQRSTUVWXYZ',
    lower: 'abcdefghijklmnopqrstuvwxyz',
    digits: '0123456789',
    symbols: '!@#$%^&*()_+-=[]{}|;:,.<>?'
  };

  const AMBIGUOUS = '0O1lI|';

  function getActiveCharsets() {
    const chips = document.querySelectorAll('#charToggles .toggle-chip');
    const active = [];
    chips.forEach(c => { if (c.classList.contains('active')) active.push(c.dataset.charset); });
    return active;

  function buildPool() {
    const includeOnly = document.getElementById('includeOnlyChars').value.trim();
    if (includeOnly) return includeOnly;

    const active = getActiveCharsets();
    let pool = active.map(k => CHARSETS[k]).join('');

    const excludeAmbiguous = document.getElementById('ambiguousToggle').classList.contains('active');
    if (excludeAmbiguous) {
      pool = pool.split('').filter(c => !AMBIGUOUS.includes(c)).join('');
    }

    const excludeChars = document.getElementById('excludeChars').value;
    if (excludeChars) {
      const excSet = new Set(excludeChars.split(''));
      pool = pool.split('').filter(c => !excSet.has(c)).join('');
    }

    return pool;
  }

  // ==================== GENERATORS ====================
  function generatePassword() {
    const pool = buildPool();
    if (!pool) return '(no characters selected)';
    const len = parseInt(document.getElementById('lengthSlider').value);
    let result = '';
    for (let i = 0; i < len; i++) {
      result += pool[secureRandom(pool.length)];
    }
    return result;
  }

  function generatePassphrase() {
    const count = parseInt(document.getElementById('wordCountSlider').value);
    const caseMode = document.getElementById('caseSelect').value;
    const sepSelect = document.getElementById('separatorSelect').value;
    const sep = sepSelect === 'custom' ? (document.getElementById('customSeparator').value || '-') : sepSelect;

    const words = [];
    for (let i = 0; i < count; i++) {
      let w = WORDLIST[secureRandom(WORDLIST.length)];
      switch (caseMode) {
        case 'title': w = w[0].toUpperCase() + w.slice(1); break;
        case 'upper': w = w.toUpperCase(); break;
        case 'lower': break;
        case 'random': w = w.split('').map(c => secureRandomFloat() > 0.5 ? c.toUpperCase() : c).join(''); break;
      }
      words.push(w);
    }

    let result = words.join(sep);

    if (document.getElementById('appendDigitToggle').classList.contains('active')) {
      result += secureRandom(10);
    }
    if (document.getElementById('appendSymbolToggle').classList.contains('active')) {
      const syms = '!@#$%^&*';
      result += syms[secureRandom(syms.length)];
    }

    return result;
  }

  function generatePronounceable() {
    const consonants = 'bcdfghjklmnprstvwz';
    const vowels = 'aeiou';
    const syllableCount = parseInt(document.getElementById('syllableSlider').value);

    let result = '';
    for (let i = 0; i < syllableCount; i++) {
      result += consonants[secureRandom(consonants.length)];
      result += vowels[secureRandom(vowels.length)];
      if (secureRandomFloat() > 0.5) {
        result += consonants[secureRandom(consonants.length)];
      }
    }

    if (document.getElementById('pronounceDigitToggle').classList.contains('active')) {
      result += secureRandom(100);
    }
    if (document.getElementById('pronounceSymbolToggle').classList.contains('active')) {
      const syms = '!@#$%^&*';
      result += syms[secureRandom(syms.length)];
    }

    return result;
  }

  function generate() {

  // ==================== ENTROPY ====================
  function calcEntropy(pw) {
    if (!pw || pw.startsWith('(')) return { bits: 0, seconds: 0 };

    if (currentMode === 'passphrase') {
      const wordCount = parseInt(document.getElementById('wordCountSlider').value);
      const bits = wordCount * Math.log2(WORDLIST.length);
      let extra = 0;
      if (document.getElementById('appendDigitToggle').classList.contains('active')) extra += Math.log2(10);
      if (document.getElementById('appendSymbolToggle').classList.contains('active')) extra += Math.log2(8);
      return { bits: bits + extra, seconds: Math.pow(2, bits + extra) / 1e9 };
    }

    if (currentMode === 'pronounce') {
      const syllables = parseInt(document.getElementById('syllableSlider').value);
      // Each syllable: consonant(18) + vowel(5) + optional consonant(50% * 18)
      const bitsPerSyll = Math.log2(18) + Math.log2(5) + 0.5 * Math.log2(18) + 1; // +1 for the choice bit
      let bits = syllables * bitsPerSyll;
      if (document.getElementById('pronounceDigitToggle').classList.contains('active')) bits += Math.log2(100);
      if (document.getElementById('pronounceSymbolToggle').classList.contains('active')) bits += Math.log2(8);
      return { bits, seconds: Math.pow(2, bits) / 1e9 };
    }

    // Password mode
    const pool = buildPool();
    if (!pool) return { bits: 0, seconds: 0 };
    const len = pw.length;
    const bits = len * Math.log2(pool.length);
    return { bits, seconds: Math.pow(2, bits) / 1e9 };
  }

  function formatCrackTime(seconds) {
    if (seconds < 1) return 'instant';
    if (seconds < 60) return '< 1 minute';
    if (seconds < 3600) return Math.floor(seconds / 60) + ' minutes';
    if (seconds < 86400) return Math.floor(seconds / 3600) + ' hours';
    if (seconds < 86400 * 365) return Math.floor(seconds / 86400) + ' days';
    if (seconds < 86400 * 365 * 100) return Math.floor(seconds / (86400 * 365)) + ' years';
    if (seconds < 86400 * 365 * 1e6) return Math.floor(seconds / (86400 * 365 * 100)) + ' centuries';
    if (seconds < 86400 * 365 * 1e9) return Math.floor(seconds / (86400 * 365 * 1e6)) + 'M years';
    if (seconds < 86400 * 365 * 1e12) return Math.floor(seconds / (86400 * 365 * 1e9)) + 'B years';
    return '> heat death of universe';
  }

  function updateStrength(pw) {
    const { bits, seconds } = calcEntropy(pw);
    const bar = document.getElementById('strengthBar');
    const pct = Math.min(100, (bits / 128) * 100);

    let color;
    if (bits < 40) color = 'var(--entropy-low)';
    else if (bits < 80) color = 'var(--entropy-mid)';
    else color = 'var(--entropy-high)';

    bar.style.width = pct + '%';
    bar.style.background = color;

    document.getElementById('entropyBits').textContent = Math.floor(bits) + ' bits';
    document.getElementById('crackTime').textContent = formatCrackTime(seconds) + ' @ 1B/s';
  }

  // ==================== PREVIEW (no history) ====================
  function updatePreview() {
    currentPreview = generate();
    document.getElementById('passwordDisplay').textContent = currentPreview;
    updateStrength(currentPreview);
  }

  // ==================== EXPLICIT GENERATE (adds to history) ====================
  function doGenerate() {
    currentPreview = generate();
    document.getElementById('passwordDisplay').textContent = currentPreview;
    updateStrength(currentPreview);
    addToHistory(currentPreview);
  }

  // ==================== COPY ====================
  async function doCopy(text) {
    try {
      await navigator.clipboard.writeText(text || currentPreview);
      const btn = document.getElementById('copyBtn');
      btn.textContent = 'Copied';
      btn.classList.add('copied');
      setTimeout(() => { btn.textContent = 'Copy'; btn.classList.remove('copied'); }, 1200);
    } catch(e) {
      // fallback
      const ta = document.createElement('textarea');
      ta.value = text || currentPreview;
      document.body.appendChild(ta);
      ta.select();
      document.execCommand('copy');
      document.body.removeChild(ta);
    }
  }

  // ==================== HISTORY ====================
  function addToHistory(pw) {
    if (!pw || pw.startsWith('(')) return;
    history.unshift(pw);
    if (history.length > 50) history.pop();
    renderHistory();
  }

  function renderHistory() {
    const list = document.getElementById('historyList');
    const section = document.getElementById('historySection');

    if (history.length === 0) {
      section.style.display = 'none';
      return;
    }

    section.style.display = 'block';
    list.innerHTML = history.map((pw, i) =>
      `<li class="history-item">
        <span class="history-item-text">${escHtml(pw)}</span>
        <button class="btn" onclick="window._copyHistory(${i})" aria-label="Copy password">Copy</button>
      </li>`
    ).join('');
  }

  window._copyHistory = function(i) {
    doCopy(history[i]);
  };

  // ==================== BATCH ====================
  function doBatchGenerate() {
    const count = Math.min(20, Math.max(2, parseInt(document.getElementById('batchCount').value) || 5));
    batchResults = [];
    for (let i = 0; i < count; i++) {
      batchResults.push(generate());
    }
    renderBatch();
  }

  function renderBatch() {
    const container = document.getElementById('batchResults');
    if (batchResults.length === 0) { container.innerHTML = ''; return; }

    let html = '<ul class="batch-list">';
    batchResults.forEach((pw, i) => {
      html += `<li class="batch-item">
        <span class="batch-item-text">${escHtml(pw)}</span>
        <button class="btn" onclick="window._copyBatch(${i})" aria-label="Copy password">Copy</button>
      </li>`;
    });
    html += '</ul>';
    html += '<button class="btn" style="margin-top:8px;width:100%" onclick="window._copyAllBatch()">Copy all</button>';
    container.innerHTML = html;
  }

  window._copyBatch = function(i) { doCopy(batchResults[i]); };
  window._copyAllBatch = function() { doCopy(batchResults.join('\n')); };

  function escHtml(s) {
    return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
  }

  // ==================== THEME ====================
  function initTheme() {
    const saved = localStorage.getItem('kin019-theme');
    if (saved === 'dark') {
      document.body.classList.add('dark');
      document.querySelector('meta[name="theme-color"]').content = '#161614';
      document.getElementById('themeToggle').innerHTML = '&#9788;';
    }
  }

  function toggleTheme() {
    const isDark = document.body.classList.contains('dark');
    if (isDark) {
      document.body.classList.remove('dark');
      document.querySelector('meta[name="theme-color"]').content = '#f5f4f0';
      document.getElementById('themeToggle').innerHTML = '&#9789;';
      localStorage.setItem('kin019-theme', 'light');
    } else {
      document.body.classList.add('dark');
      document.querySelector('meta[name="theme-color"]').content = '#161614';
      document.getElementById('themeToggle').innerHTML = '&#9788;';
      localStorage.setItem('kin019-theme', 'dark');
    }
  }

  // ==================== MODE SWITCHING ====================
  function setMode(mode) {
    currentMode = mode;
    document.querySelectorAll('.mode-tab').forEach(t => t.classList.toggle('active', t.dataset.mode === mode));
    document.getElementById('passwordControls').classList.toggle('hidden', mode !== 'password');
    document.getElementById('passphraseControls').classList.toggle('hidden', mode !== 'passphrase');
    document.getElementById('pronounceControls').classList.toggle('hidden', mode !== 'pronounce');
    document.getElementById('customCharSection').classList.toggle('hidden', mode !== 'password');
    updatePreview();
  }

  // ==================== EXPANDABLE SECTIONS ====================
  function setupExpandable(triggerId, bodyId) {
    const trigger = document.getElementById(triggerId);
    const body = document.getElementById(bodyId);
    trigger.addEventListener('click', () => {
      trigger.classList.toggle('open');
      body.classList.toggle('open');
    });
  }

  // ==================== TOGGLE CHIPS ====================
  function setupToggleChip(el) {
    el.addEventListener('click', () => {
      // For charset toggles, ensure at least one remains active
      if (el.parentElement.id === 'charToggles') {
        const siblings = el.parentElement.querySelectorAll('.toggle-chip.active');
        if (el.classList.contains('active') && siblings.length <= 1) return;
      }
      el.classList.toggle('active');
      el.setAttribute('aria-pressed', el.classList.contains('active'));
      updatePreview();
    });
  }

  // ==================== INIT ====================
  function init() {
    initTheme();

    // Theme toggle
    document.getElementById('themeToggle').addEventListener('click', toggleTheme);

    // Mode tabs
    document.querySelectorAll('.mode-tab').forEach(tab => {
      tab.addEventListener('click', () => setMode(tab.dataset.mode));
    });

    // Generate / Copy
    document.getElementById('generateBtn').addEventListener('click', doGenerate);
    document.getElementById('copyBtn').addEventListener('click', () => doCopy());

    // Sliders - preview only, no history
    document.getElementById('lengthSlider').addEventListener('input', (e) => {
      document.getElementById('lengthValue').textContent = e.target.value;
      updatePreview();
    });

    document.getElementById('wordCountSlider').addEventListener('input', (e) => {
      document.getElementById('wordCountValue').textContent = e.target.value;
      updatePreview();
    });

    document.getElementById('syllableSlider').addEventListener('input', (e) => {
      document.getElementById('syllableValue').textContent = e.target.value;
      updatePreview();
    });

    // Char toggles
    document.querySelectorAll('#charToggles .toggle-chip').forEach(setupToggleChip);

    // Simple toggles
    ['ambiguousToggle', 'appendDigitToggle', 'appendSymbolToggle', 'pronounceDigitToggle', 'pronounceSymbolToggle'].forEach(id => {
      const el = document.getElementById(id);
      el.addEventListener('click', () => {
        el.classList.toggle('active');
        el.setAttribute('aria-pressed', el.classList.contains('active'));
        updatePreview();
      });
    });

    // Separator select
    document.getElementById('separatorSelect').addEventListener('change', (e) => {
      document.getElementById('customSeparator').classList.toggle('hidden', e.target.value !== 'custom');
      updatePreview();
    });

    document.getElementById('customSeparator').addEventListener('input', updatePreview);
    document.getElementById('caseSelect').addEventListener('change', updatePreview);

    // Custom chars
    document.getElementById('excludeChars').addEventListener('input', updatePreview);
    document.getElementById('includeOnlyChars').addEventListener('input', updatePreview);

    // Expandables
    setupExpandable('customCharTrigger', 'customCharBody');
    setupExpandable('batchTrigger', 'batchBody');

    // Batch generate
    document.getElementById('batchGenBtn').addEventListener('click', doBatchGenerate);

    // Clear history
    document.getElementById('clearHistoryBtn').addEventListener('click', () => {
      history = [];
      renderHistory();
    });

    // Keyboard shortcuts
    document.addEventListener('keydown', (e) => {
      // Don't fire when typing in inputs
      if (e.target.tagName === 'INPUT' || e.target.tagName === 'TEXTAREA' || e.target.tagName === 'SELECT') return;

      if (e.code === 'Space') {
        e.preventDefault();
        doGenerate();
      } else if (e.code === 'Enter') {
        e.preventDefault();
        doCopy();
      }
    });

    // Hide history initially
    document.getElementById('historySection').style.display = 'none';

    // Initial preview
    updatePreview();
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }

})();
</script>
